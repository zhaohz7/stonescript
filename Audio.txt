//private Output func with toggler
var check = import UI\Toggler
func Output(x, y, color, a)
  ?check.toggler()
    >`@x@,@y@,@color@,@a@
//
func roll( x, y, scroll, limit, inputindex)
  ?loc.begin
   ?inputindex = storage.Get("latestinput",inputindex)
    index = storage.Get("indexmemory", inputindex)
   :
    index = inputindex
  storage.Set("indexmemory", index)
  storage.Set("latestinput", inputindex)
  ?timer>0
   timer--
  :?timer<0
   timer++  
  
  ?key=down|key=up
   ?accel<scroll*30
    accel++
  
  ?key=upEnd|key=downEnd
   accel=0
  
  ?key=down&timer<=0
   timer=scroll-accel/30
   index--
   ?index<0
    index=soundlist.Count()-1
  
  ?key=up&timer<=0
   timer=scroll-accel/30
   index++
   ?index>=soundlist.Count()
    index=0
  
  ?index > 9
    Output(x,y,#ffffff,index)
  :
    Output(x+1,y,#ffffff,index)
  
  Output(x+4,y,#ffffff,soundlist[index])
  ?index>0&index<soundlist.Count()-1
    Output(x+1,y-1,#666666,"▲  "+soundlist[index+1])
    Output(x+1,y+1,#666666,"▼  "+soundlist[index-1])
  :?index=0
    Output(x+1,y-1,#666666,"▲  "+soundlist[index+1])
    Output(x+1,y+1,#666666,"▼  "+soundlist[soundlist.Count()-1])
  :?index=soundlist.Count()-1
    Output(x+1,y-1,#666666,"▲  "+soundlist[0])
    Output(x+1,y+1,#666666,"▼  "+soundlist[index-1])
  
  ?key=left&key=right&timer>=0
   timer=limit
   for i = 0..soundlist.Count()-1
     ?i = index
       play @soundlist[index]@
